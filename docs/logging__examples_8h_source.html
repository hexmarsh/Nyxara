52 void CustomFormatterExample()
53 {
54 using namespace Nyxara::Logging;
55
56 // Register a JSON formatter
57 ImprovedLogger::SetFormatter([](const LogEntry& entry) -> std::string {
58 std::ostringstream json;
59 json << "{";
60 json << "\"timestamp\":\"" << std::chrono::duration_cast<std::chrono::milliseconds>(
61 entry.GetTimestamp().time_since_epoch()).count() << "\",";
62 json << "\"level\":\"" << static_cast<int>(entry.GetLevel()) << "\",";
63 json << "\"category\":\"" << entry.GetCategory().GetName() << "\",";
64 json << "\"message\":\"" << entry.GetMessage() << "\"";
65
66 if (entry.GetCallDepth() >= 0)
67 {
68 json << ",\"call_depth\":" << entry.GetCallDepth();
69 }
70
71 if (entry.HasMetadata())
72 {
73 json << ",\"metadata\":{";
74 bool first = true;
75 for (const auto& [key, value] : entry.GetMetadata())
76 {
77 if (!first) json << ",";
78 first = false;
79 json << "\"" << key << "\":";
80 std::visit([&json](const auto& v) {
81 using T = std::decay_t<decltype(v)>;
82 if constexpr (std::is_same_v<T, std::string>)
83 {
84 json << "\"" << v << "\"";
85 }
86 else
87 {
88 json << v;
89 }
90 }, value);
91 }
92 json << "}";
93 }
94
95 json << "}";
96 return json.str();
97 });
98
99 // Log something
100 auto entry = ImprovedLogger::CreateEntry(ExampleCategory, Verbosity::Info)
101 .Message("API request processed")
102 .WithMetadata("endpoint", std::string("/api/users"))
103 .WithMetadata("response_time_ms", int64_t(250))
104 .Build();
105
106 ImprovedLogger::Log(std::move(entry));
107 // Output: {"timestamp":"1234567890","level":"3","category":"ExampleCategory","message":"API request processed","call_depth":0,"metadata":{"endpoint":"/api/users","response_time_ms":250}}
108
109 // Clear custom formatter to return to default
110 ImprovedLogger::ClearFormatter();
111 }

116 void FilteringExample()
117 {
118 using namespace Nyxara::Logging;
119
120 // Add a filter to suppress logs containing "sensitive"
121 ImprovedLogger::AddFilter("sensitive_filter", [](const LogEntry& entry) -> bool {
122 return entry.GetMessage().find("sensitive") == std::string::npos;
123 });
124
125 // Add a filter to only allow logs with specific metadata
126 ImprovedLogger::AddFilter("admin_only", [](const LogEntry& entry) -> bool {
127 if (!entry.HasMetadata()) return true;
128
129 auto& metadata = entry.GetMetadata();
130 auto it = metadata.find("is_admin");
131 if (it != metadata.end())
132 {
133 return std::get<bool>(it->second);
134 }
135 return true; // Allow if no is_admin metadata
136 });
137
138 // This will be filtered out
139 NYX_LOG_INFO(ExampleCategory, "This contains sensitive information");
140
141 // This will also be filtered out
142 auto entry1 = ImprovedLogger::CreateEntry(ExampleCategory, Verbosity::Info)
143 .Message("User action")
144 .WithMetadata("is_admin", false)
145 .Build();
146 ImprovedLogger::Log(std::move(entry1));
147
148 // This will pass through
149 auto entry2 = ImprovedLogger::CreateEntry(ExampleCategory, Verbosity::Info)
150 .Message("Admin performed action")
151 .WithMetadata("is_admin", true)
152 .Build();
153 ImprovedLogger::Log(std::move(entry2));
154
155 // Clean up filters
156 ImprovedLogger::RemoveFilter("sensitive_filter");
157 ImprovedLogger::RemoveFilter("admin_only");
158 }
